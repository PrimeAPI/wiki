## Import (Maven)

```xml
    <repositories>
        <repository>
            <url>https://repo.primeapi.de</url>
            <id>PrimeAPI</id>
        </repository>
    </repositories>

   <dependency>
       <groupId>de.primeapi.primeplugins</groupId>
       <artifactId>SpigotAPI</artifactId>
       <version>LATEST</version>
       <scope>provided</scope>
   </dependency>
```
